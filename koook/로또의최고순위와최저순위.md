def solution(lottos, win_nums):
    answer = []

    iNumberOfZero = lottos.count(0) # 0개수 확인

        if (i in win_nums):
            iNumberOfCouract += 1

    answer = [Grad[iNumberOfCouract + iNumberOfZero], Grad[iNumberOfCouract]]
    
    return answer
